    # 回溯算法！

    • 组合问题：N 个数里面按一定规则找出 k 个数的集合
    • 切割问题：一个字符串按一定规则有几种切割方式
    • 子集问题：一个 N 个数的集合里有多少符合条件的子集
    • 排列问题：N 个数按一定规则全排列，有几种排列方式
    • 棋盘问题：N 皇后，解数独等等
    • 回溯法解决的问题都可以抽象为树形结构，是的，我指的是所有回溯法的问题都可以抽象为树形结构！因为回溯法解决的都是在集合中递归查找子集，集合的大小就构成了树的宽度，递归的深度，都构成的树的深度。递归就要有终止条件，所以必然是一棵高度有限的树（N 叉树）。
    • 组合是不强调元素顺序的，排列是强调元素顺序。

    • var combinationSum2 = function (candidates, target) {
  let path = [];
  let res = [];
  candidates.sort((a, b) => a - b);

  const fn = function (index, sum) {
    if (sum === target) {
      res.push(path.slice());
      return;
    }
    for (let i = index; i < candidates.length; i++) {
      if (candidates[i] > target - sum) return;
      if (i > index && candidates[i] == candidates[i - 1]) continue;
      path.push(candidates[i]);
      fn(i + 1, candidates[i] + sum);
      path.pop();
    }
  };
  fn(0, 0);
  return res;
};

    • var permuteUnique = function (nums) {

      used[i] = true;
      fn();
      used[i] = false;
      path.pop();
    }
  };
  fn();
  return res;
};

    • var solveNQueens = function (n) {
  let res = [];
  let arr = new Array(n).fill().map(() => new Array(n).fill("."));
  const fn1 = function (row, col) {
    //竖直方向
    for (let i = 0; i < row; i++) {
      if (arr[i][col] == "Q") return false;
    }

    //左斜
    for (let i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {
      if (arr[i][j] == "Q") return false;
    }
    //右斜
    for (let i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
      if (arr[i][j] == "Q") return false;
    }
    return true;
  };
  const fn = function (row) {
    if (row === n) {
      let cur = [];
      arr.forEach((item) => {
        let str = "";
        item.forEach((item1) => {
          str += item1;
        });
        cur.push(str);
      });
      res.push(cur);
      return;
    }
    for (let i = 0; i < n; i++) {
      if (fn1(row, i)) {
        arr[row][i] = "Q";
        fn(row + 1);
        arr[row][i] = ".";
      }
    }
  };
  fn(0);
  return res;
};

    let startCol = Math.floor(col / 3) * 3;

    for (let i = startRow; i < startRow + 3; i++) {
      for (let j = startCol; j < startCol + 3; j++) {
        if (board[i][j] === val) {
          return false;
        }
      }
    }

    return true;
  }
  function fn() {
    for (let i = 0; i < board.length; i++) {
      for (let j = 0; j < board[0].length; j++) {
        if (board[i][j] !== ".") continue;
        for (let val = 1; val <= 9; val++) {
          if (isValid(i, j, val + "", board)) {
            board[i][j] = val.toString();
            if (fn()) {
              return true;
            }

            board[i][j] = `.`;
          }
        }
        return false;
      }
    }
    return true;
  }
  fn();
  return board;
};
