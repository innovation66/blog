    # 1.每日温度

    • https://leetcode.cn/problems/daily-temperatures/ (opens new window)

    • 就是找当天之后的第一个更高温度的值，使用栈结构，栈中保存从大到小的序列，如果新加入进来的值大于 top 值 开始往外 pop ， 直到小于 top，这个新的值就是排出来的这几天的更高温度！

    • var dailyTemperatures = function (temperatures) {
  let res = [];
  let stack = [0];
  for (let index = 1; index < temperatures.length; index++) {
    while (
      stack.length &&
      temperatures[index] > temperatures[stack[stack.length - 1]]
    ) {
      let item = stack.pop();
      console.log(item);
      res[item] = index - item;
    }
    stack.push(index);
  }
  while (stack.length) {
    let item = stack.pop();
    res[item] = 0;
  }
  return res;
};

  let stack = [0];

    # 3.下一个更大元素 2

    • https://leetcode.cn/problems/next-greater-element-ii/ (opens new window)

    • 数组变成了循环数组，别的也没啥区别，只需要在 index 的映射问题上处理好，数组不需要加长，for 循环的 length 加长即可

    • var nextGreaterElements = function (nums) {
  const len = nums.length;
  let stack = [];
  let res = Array(len).fill(-1);
  for (let i = 0; i < len * 2; i++) {
    while (stack.length && nums[i % len] > nums[stack[stack.length - 1]]) {
      const index = stack.pop();
      res[index] = nums[i % len];
    }
    stack.push(i % len);
  }
  return res;
};

    • 还有一种方法就是使用单调栈，上面找最大值 使用的是竖着看 一条一条的加在一起。使用单调栈

      便是横向看，遇到比 top 更大的值 pop 然后 top 和新值就是 pop 出来的值的左右两边 取他们中较小的那个减去本身的高度 作为高 再算出他们之间的距离。

    • let stack = [0];
let res = 0;
for (let i = 1; i < height.length; i++) {
  while (stack.length && height[i] > height[stack[stack.length - 1]]) {
    let index = stack.pop();
    if (stack.length) {
      let h =
        Math.min(height[stack[stack.length - 1]], height[i]) - height[index];
      let w = i - stack[stack.length - 1] - 1;
      res += h * w;
    }
  }
  stack.push(i);
}
return res;

    # 5.柱状图最大矩形

    • https://leetcode.cn/problems/largest-rectangle-in-histogram/description/ (opens new window)

    • 这道题和接雨水有点类似 也是两种方法 第一个就是找出一个矩形左右两侧的最远的大于他的值 然后相减再乘上他的值 第二个就是单调栈 栈中存放的是单调递增的值 遇到小的就开始 pop，分别计算以 pop 出的值为高度的矩形面积 为了保障全部矩形都被计算到 上代码！在第一种方法，求两面最大 是可递归的 ！

    • var largestRectangleArea = function (heights) {
  //     heights = [0,...heights,0]
  //    let stack = []
  //     let max = 0
  //     for(let i = 0 ; i < heights.length ; i++){
  //         while(stack.length&&heights[i]<heights[stack[stack.length-1]]){
  //             let index = stack.pop()
  //             let w = i - stack[stack.length -1] - 1;
  //             let h = heights[index]
  //             max = Math.max(max,w*h)
  //         }
  //         stack.push(i)
  //     }
  //     return max
  let left = new Array(heights.length);
  let right = new Array(heights.length);
  left[0] = -1;
  right[heights.length - 1] = heights.length;
  for (let i = 1; i < heights.length; i++) {
    let cur = i - 1;
    while (heights[cur] >= heights[i] && cur >= 0) {
      cur = left[cur];
    }
    left[i] = cur;
  }
  for (let i = heights.length - 2; i >= 0; i--) {
    let t = i + 1;
    // 这里不是用if，而是不断向右寻找的过程
    while (t < heights.length && heights[t] >= heights[i]) t = right[t];
    right[i] = t;
  }
  let max = 0;
  for (let i = 0; i < heights.length; i++) {
    let sum = heights[i] * (right[i] - left[i] - 1);
    max = Math.max(max, sum);
  }

  return max;
};
